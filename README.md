```xml
    <repositories>
        <repository>
            <id>jitpack.io</id>
            <url>https://jitpack.io</url>
        </repository>
    </repositories>

    <dependencies>
        <dependency>
            <groupId>com.github.e3ndr</groupId>
            <artifactId>EndersUtil</artifactId>
            <version>1.3.0</version>
        </dependency>
    </dependencies>
```
